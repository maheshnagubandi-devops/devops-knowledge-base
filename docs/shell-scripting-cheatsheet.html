Error Handling
3
Script Safety
#!/bin/bash set -euo pipefail # -e: exit on error # -u: unset vars error # -o pipefail: pipe fails if any part fails
  • set -x — debug/trace
  • Always quote variables
  • chmod 755 not 777
  • Use ./script.sh
Argument Parsing
while getopts ":u:p:h" opt; do case $opt in u) user="$OPTARG" ;; p) pass="$OPTARG" ;; h) echo "help"; exit 0 ;; \?) echo "Invalid" ;; esac done
Trap & Signals
trap cleanup EXIT INT TERM cleanup() { rm -f /tmp/tempfile echo "Cleaned up" } # Signal handlers trap 'echo "CTRL+C"; exit 2' INT
Read Input
read -p "Name: " name read -s -p "Pass: " pass read -n 1 -p "Y/N: " choice read -t 5 -p "5s timeout: " input
  • -p — prompt
  • -s — silent (pwd)
  • -t — timeout

Common Patterns
# Check if arg exists [ -z "$1" ] && { echo "Need arg"; exit 1; } # Check if file exists [ -f "$file" ] || touch "$file" # Loop over files find . -name "*.log" | while read f; do echo "$f" done # Safe temp file tmpfile=$(mktemp) trap "rm -f $tmpfile" EXIT # Check if running as root [ "$EUID" -ne 0 ] && { echo "Need root"; exit 1; } # Command success/failure cmd_that_might_fail || true grep pattern file || echo "Not found"

Quick Tips: Always quote variables ($var). Use [[ ]] for bash-only. Test files before modifying. Use trap for cleanup. Add -x to debug. Use || true to ignore failures in CI/CD.
